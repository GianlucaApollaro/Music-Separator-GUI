2. Patch patchedloadmodel - ramo FNO
Sostituisci il blocco elif hasfno: con questo:
elif hasfno:
    log.info(f"Patch FNO architecture detected for {ckpt_filename}. Instantiating full FNO1d MaskEstimator.")
    
    # Recupera parametri FNO dal YAML o fallback standard
    fno_depth = model_section.get('mask_estimator_depth', 2)  # o specifica nel YAML
    fno_dim = model_section.get('dim', 512)
    
    # Crea MaskEstimator FNO custom
    from neuraloperator import FNO1d  # Import dinamico
    class FNOMaskEstimator(MaskEstimator):
        def __init__(self, dim, dim_inputs, depth, mlp_expansion_factor):
            super().__init__(dim, dim_inputs, depth, mlp_expansion_factor)
            # Sostituisci il tuo MLP con FNO1d
            self.fno_layers = nn.ModuleList([
                FNO1d(
                    modes=12,  # parametri dal repo pcunwa
                    channels=dim,
                    layers=depth,
                    hidden_channels=dim * mlp_expansion_factor
                ) for _ in range(depth)
            ])
        
        def forward(self, x):  # semplificato - adatta alla firma MaskEstimator
            for fno in self.fno_layers:
                x = fno(x) + self.dropout(x)  # skip connection
            return torch.sigmoid(x)  # o l'output atteso dal mask estimator
    
    # Ricostruisci modello con FNO estimator
    model_args['mask_estimator'] = FNOMaskEstimator(
        model_args['dim'], 
        model_args['dim'], 
        model_args.get('mask_estimator_depth', 2),
        model_args.get('mlp_expansion_factor', 4)
    )
    
    if is_bs:
        model = BSRoformer(model_args)
    else:
        model = MelBandRoformer(model_args)
    
    # Load strict=True ora funziona perché hai i layer FNO
    model.load_state_dict(sd, strict=True)
